221 #define DBL_LINKED_LIST_ADD(l,e) \
222  do { if ((l)) (l)->prev = (e); \
223  (e)->next = (l); \
224  (l) = (e); \
225  } while(0)
226 
227 #define DBL_LINKED_LIST_REMOVE(l,e) \
228  do { if ((l) == (e)) (l) = (e)->next; \
229  if ((e)->prev) (e)->prev->next = (e)->next; \
230  if ((e)->next) (e)->next->prev = (e)->prev; \
231  } while(0)
