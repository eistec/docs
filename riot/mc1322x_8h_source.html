401 /* Macro to safely disable all interrupts for a block of code.
402  Use it like this:
403  disable_int({
404  asdf = 1234;
405  printf("hi\r\n");
406  });
407 */
408 #define __int_top() volatile uint32_t saved_intenable
409 #define __int_disable() saved_intenable = ITC->INTENABLE; ITC->INTENABLE = 0
410 #define __int_enable() ITC->INTENABLE = saved_intenable
411 #define disable_int(x) do { \
412  __int_top(); \
413  __int_disable(); \
414  x; \
415  __int_enable(); } while(0)
