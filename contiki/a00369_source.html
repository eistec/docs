57 void
58 shell_default_output(const char *text1, int len1, const char *text2, int len2)
59 {
60  int i;
61  if(text1 == NULL) {
62  text1 = "";
63  len1 = 0;
64  }
65  if(text2 == NULL) {
66  text2 = "";
67  len2 = 0;
68  }
69 
70  /* Precision (printf("%.Ns", text1)) not supported on all platforms.
71  putchar(c) not be supported on all platforms. */
72  for(i = 0; i < len1; i++) {
73  printf("%c", text1[i]);
74  }
75  for(i = 0; i < len2; i++) {
76  printf("%c", text2[i]);
77  }
78  printf("\r\n");
79 }

93 PROCESS_THREAD(serial_shell_process, ev, data)
94 {
95  PROCESS_BEGIN();
96 
97  shell_init();
98 
99  while(1) {
100  PROCESS_WAIT_EVENT_UNTIL(ev == serial_line_event_message && data != NULL);
101  shell_input(data, strlen(data));
102  }
103 
104  PROCESS_END();
105 }
