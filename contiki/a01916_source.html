68 static void
69 input(void)
70 {
71  if(packetbuf_datalen() > 0 &&
72  packetbuf_datalen() <= UIP_BUFSIZE - UIP_LLH_LEN) {
73  memcpy(&uip_buf[UIP_LLH_LEN], packetbuf_dataptr(), packetbuf_datalen());
74  uip_len = packetbuf_datalen();
75  tcpip_input();
76  }
77 }
