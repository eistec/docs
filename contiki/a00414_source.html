49 PROCESS(shell_reboot_process, "reboot");
50 SHELL_COMMAND(reboot_command,
51  "reboot",
52  "reboot: reboot the system",
53  &shell_reboot_process);
54 /*---------------------------------------------------------------------------*/
55 PROCESS_THREAD(shell_reboot_process, ev, data)
56 {
57  static struct etimer etimer;
58 
59  PROCESS_EXITHANDLER(leds_off(LEDS_ALL);)
60 
61  PROCESS_BEGIN();
62 
63  shell_output_str(&reboot_command,
64  "Rebooting the node in four seconds...", "");
65 
66  etimer_set(&etimer, CLOCK_SECOND);
67  PROCESS_WAIT_UNTIL(etimer_expired(&etimer));
68  leds_on(LEDS_RED);
69  etimer_reset(&etimer);
70  PROCESS_WAIT_UNTIL(etimer_expired(&etimer));
71  leds_on(LEDS_GREEN);
72  etimer_reset(&etimer);
73  PROCESS_WAIT_UNTIL(etimer_expired(&etimer));
74  leds_on(LEDS_BLUE);
75  etimer_reset(&etimer);
76  PROCESS_WAIT_UNTIL(etimer_expired(&etimer));
77 
78  watchdog_reboot();
79 
80  PROCESS_END();
81 }
82 /*---------------------------------------------------------------------------*/
83 void
84 shell_reboot_init(void)
85 {
86  shell_register_command(&reboot_command);
87 }
