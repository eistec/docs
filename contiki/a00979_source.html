52 /* the typical clock things like incrementing current_clock and etimer checks */
53 /* are performed as a periodically scheduled rtimer */
54 void
55 rt_do_clock(struct rtimer *t, void *ptr)
56 {
57  rtimer_set(t, RTIMER_TIME(t) + (rtc_freq/CLOCK_CONF_SECOND) , 1,
58  (rtimer_callback_t)rt_do_clock, ptr);
59 
60  current_clock++;
61 
62  if((current_clock % CLOCK_CONF_SECOND) == 0) {
63  seconds++;
64  }
65 
66  if(etimer_pending() &&
67  (etimer_next_expiration_time() - current_clock - 1) > MAX_TICKS) {
68  etimer_request_poll();
69  }
70 
71 }

100 /**
101  * Delay the CPU for up to 65535 microseconds.
102  * Use the 250KHz MACA clock for longer delays to avoid interrupt effects.
103  * However that can't be used if the radio is being power cycled!
104  */
105 void
106 clock_delay_usec(uint16_t howlong)
107 {
108  if(howlong<2) return;
109 #if 0
110  if(howlong>400) {
111  volatile register uint32_t i=*MACA_CLK+howlong/4;
112  while (i > *MACA_CLK) ;
113  return;
114  }
115 #endif
116  /* These numbers at 25MHz, gcc -Os */
117  volatile register uint32_t i=4000*howlong/2301;
118  while(--i);
119 }
120 /*---------------------------------------------------------------------------*/
121 /**
122  * Delay the CPU for up to 65535 milliseconds. The watchdog is NOT disabled.
123  */
124 void
125 clock_delay_msec(uint16_t howlong)
126 {
127  while(howlong--) clock_delay_usec(1000);
128 }
129 /*---------------------------------------------------------------------------*/
130 /**
131  * Legacy delay. The original clock_delay for the msp430 used a granularity
132  * of 2.83 usec. This approximates that delay for values up to 1456 usec.
133  * (The largest core call in leds.c uses 400).
134  */
135 void
136 clock_delay(unsigned int howlong)
137 {
138  if(howlong--) return;
139  clock_delay_usec((283*howlong)/100);
140 }

142 /**
143  * Adjust clock ticks after a cpu sleep.
144  */
145 void clock_adjust_ticks(clock_time_t howmany) {
146 /* Add seconds */
147  seconds+=howmany/CLOCK_CONF_SECOND;
148 /* Handle tick overflow */
149  if(((current_clock % CLOCK_CONF_SECOND) + (howmany % CLOCK_CONF_SECOND)) >= CLOCK_CONF_SECOND) seconds++;
150 /* Add ticks */
151  current_clock+=howmany;
152 }
