46 /**
47  * Initialize debug UART used by printf.
48  *
49  * \note Undefining BOARD_DEBUG_UART_TX_PIN_PORT will disable printf.
50  */
51 void
52 dbg_uart_init(void)
53 {
54 #ifdef BOARD_DEBUG_UART_TX_PIN_PORT
55  /* Enable the clock gate to the TX pin PORT */
56  port_module_enable(BOARD_DEBUG_UART_TX_PIN_PORT);
57  /* Choose UART TX for the pin mux and disable PORT interrupts on the pin */
58  BOARD_DEBUG_UART_TX_PIN_PORT->PCR[BOARD_DEBUG_UART_TX_PIN_NUMBER] =
59  PORT_PCR_MUX(BOARD_DEBUG_UART_TX_PIN_MUX);
60 #endif
61 #ifdef BOARD_DEBUG_UART_RX_PIN_PORT
62  port_module_enable(BOARD_DEBUG_UART_RX_PIN_PORT);
63 
64  /* Choose UART RX for the pin mux and disable PORT interrupts on the pin */
65  BOARD_DEBUG_UART_RX_PIN_PORT->PCR[BOARD_DEBUG_UART_RX_PIN_NUMBER] =
66  PORT_PCR_MUX(BOARD_DEBUG_UART_RX_PIN_MUX);
67 #endif
68 
69  uart_init(BOARD_DEBUG_UART_NUM, 0, BOARD_DEBUG_UART_BAUD);
70 }
