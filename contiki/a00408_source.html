45 PROCESS(shell_powertrace_process, "powertrace");
46 SHELL_COMMAND(powertrace_command,
47  "powertrace",
48  "powertrace [interval]: turn powertracing on or off, with reporting interval <interval>",
49  &shell_powertrace_process);
50 /*---------------------------------------------------------------------------*/
51 PROCESS_THREAD(shell_powertrace_process, ev, data)
52 {
53  char buf[10];
54  int interval;
55 
56  PROCESS_BEGIN();
57 
58  interval = shell_strtolong(data, NULL);
59 
60  if(data == NULL || interval == 0) {
61  powertrace_stop();
62  powertrace_print("");
63  } else {
64  powertrace_start(interval * CLOCK_SECOND);
65  sprintf(buf, "%d", interval);
66  shell_output_str(&powertrace_command, "Starting powertrace with interval ", buf);
67  }
68  PROCESS_END();
69 }
70 /*---------------------------------------------------------------------------*/
71 void
72 shell_powertrace_init(void)
73 {
74  shell_register_command(&powertrace_command);
75  powertrace_sniff(POWERTRACE_ON);
76 }
