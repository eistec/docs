40 /* The enhanced Apple//e doesn't have a hardware clock whatsoever. Therefore the
41  * cc65 C-library for this target doesn't include an implementation of clock().
42  */
43 
44 static unsigned char tick = 14;
45 static clock_time_t time;
46 
47 /*-----------------------------------------------------------------------------------*/
48 void
49 clock_init(void)
50 {
51  if(get_ostype() >= APPLE_IIGS) {
52  if(*(signed char *)0xC036 < 0) {
53  /* 5 / 14 = 1.0MHz / 2.8MHz */
54  tick = 5;
55  log_message("Assuming 2.8 MHz ...", "");
56  }
57  }
58 }
59 /*-----------------------------------------------------------------------------------*/
60 void
61 clock_update(void)
62 {
63  static unsigned int count;
64 
65  count += tick;
66  if(count > 2000) {
67  count = 0;
68  ++time;
69  }
70 }
71 /*-----------------------------------------------------------------------------------*/
72 clock_time_t
73 clock_time(void)
74 {
75  return time;
76 }

78 unsigned long
79 clock_seconds(void)
